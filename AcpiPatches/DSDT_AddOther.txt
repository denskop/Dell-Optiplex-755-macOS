# Add Intel ME device
into device name_hid PNP0A08 insert
begin
Device (HECI)\n
{\n
    Name (_ADR, 0x00030000)\n
}\n
end;

# Add DRAM device
into device name_hid PNP0A08 insert
begin
Device (ESI)\n
{\n
    Name (_ADR, Zero)\n
    Method (_STA, 0, NotSerialized)\n
    {\n
        Return (0x0B)\n
    }\n
}\n
end;
